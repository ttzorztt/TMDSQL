# 多线程学习记录
## `<thread>`
```C++
#include <thread>
void funa() {
  cout << "fun()" << endl;
}
void funb(int x) {
  cout << "funb(int)" << endl;
}
void func(int& x) {
  cout << "func(int&)" << endl;
}
void fund(int* x) {
  cout << "fund(int*)" << endl;
}
int main(){
  int x = 10;
  std::thread tha(funa);
  std::thread thb(funb, x);
  std::thread thc(func, std::ref(x));  // 传引用必须使用std::ref申明
  std::thread thd(fund, &x);

  thb.join();
  tha.join();
  thc.join();
  thd.join();
    return 0;
}
```
